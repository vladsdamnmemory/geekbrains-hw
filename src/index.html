    function createArrByLength(n) {
        // a. Написать функцию, которая принимает в качестве параметра число n.
        // Результатом работы функции является массив из N элементов со значениями 1, 2, 3… n.
        if (typeof n !== "number") throw new TypeError("Argument is not a number");
        let i = 0, arr = [];
        while (i < n) {
            arr.push(++i)
        }
        return arr;
    }

    function reduce(arr) {
        // b. Написать функцию, которая принимает массив чисел. Результатом работы функции является сумма чисел этого массива.
        if (!arr.every(value => typeof value === "number")) throw new TypeError("Array must contain numbers only");
        return arr.reduce((acc, val) => acc += val);
    }

    function getMax(arr) {
        //  c. Написать функцию, которая на вход получает массив целых чисел, и в качестве результата возвращает максимальное число.
        if (!arr.every(value => typeof value === "number" && value % 1 === 0)) throw new TypeError("Array must contain digits");
        return Math.max(...arr);
    }

    function getMin(arr) {
        //     d. Написать функцию, которая на вход получает массив целых чисел, и в качестве результата возвращает минимальное число.
        if (!arr.every(value => typeof value === "number" && value % 1 === 0)) throw new TypeError("Array must contain digits");
        return Math.min(...arr);
    }

    function getEvenArr(arr) {
        if (!arr.every(value => typeof value === "number" && value % 1 === 0)) throw new TypeError("Array must contain digits");
        //     f. Написать функцию, которая на вход получает массив целых чисел, и в качестве результата возвращает только четные числа из этого массива.
        //     Чтобы определить четность числа, воспользуйтесь оператором для подсчета остатка от деления: x % 2. Если остаток от деления числа на 2 равен 0,
        //     число будет четное.
        return arr.filter(v => v % 2 === 0);
    }

    function blindPrint(attempt = 0) {

        let
            options = [["ф", "а"], ["ы", "в"], ["о", "ж"]],
            match = '',
            limitIndex = 2, // Можно добавлять хоть сколько попыток
            elem = document.createElement("div");

        for (let i = 0; i < 10; ++i) {
            match += options[attempt][Math.floor(Math.random() * 2)];
        }


        alert("LEVEL #" + (attempt + 1) + "\n\nПоставьте мизинец левой руки на букву Ф, безымянный палец — на Ы, средний — на В, указательный — на А. " +
            "\nМизинец правой руки на букву Ж, безымянный палец — на Д, средний — на Л, указательный — на О. " +
            "\n\nЗапомните расположение пальцев");


        elem.classList.add("overlay");
        elem.innerHTML = `< h3 >Введите, не глядя на клавиатуру:< /h3 >< p >${match}< /p >< div >< input id='val' maxlength="10" name='val'/>< div style="text-align: center; padding-top: 12px;">< button >Отмена< /button >< /div >< /div >`
        document.body.appendChild(elem);
        elem.querySelector('button').addEventListener('click', removeElement, false);
        document.getElementById("val").addEventListener('input', onInput, false)
        document.getElementById("val").focus();

        function removeElement() {
            elem.parentNode.removeChild(elem);
        }

        function onInput(ev) {
            let value = ev.target.value;

            if (value.length === 10) {
                removeElement();

                setTimeout(() => {
                    if (value === match) {
                        alert("Все верно!");
                        attempt < limitIndex ? blindPrint(++attempt) : alert('ПОБЕДА!');
                    } else {
                        alert("Ошибка. Это конец!");
                    }
                }, 0);
            }
        }
    }
