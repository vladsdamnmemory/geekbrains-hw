    function calculateArea() {
        // Пользователь вводит длину оснований трапеции (a и b), а также высоту трапеции h.
        // Программа выводит сообщение: «Площадь трапеции будет равна значение».
        // Площадь вычисляется по формуле ((a + b) / 2) * h, где a, b - основания, h - высота.
        let h = parseInt(prompt("Pass the height", "10"));
        let a = parseInt(prompt("Pass the 'a' length", "3"));
        let b = parseInt(prompt("Pass the 'b' length", "5"));

        let result = ((a + b) / 2) * h;

        if (!isNaN(result)) {
            alert(`Trapeze area equals: ${result}`)
        }
    }

                // Success
                alert("Верно!");

                if (++wasCorrect - i === 1) {
                    alert(`Да Вы знаток! Добавьтесь в телеграмме 🥺`);
                    alert("Вы победили в этой игре! Больше загадок не будет!");

                }
            }

            if (i === questions.length - 1) {
                alert(`Разгадано загадок: ${wasCorrect}`);
                alert(`Game over!`);
            }

        }


        function shuffle(arr) {
            let j, x, i;
            for (i = arr.length - 1; i > 0; i--) {
                j = Math.floor(Math.random() * (i + 1));
                x = arr[i];
                arr[i] = arr[j];
                arr[j] = x;
            }
            return arr;
        }

    }
